```cpp
std::size_t hash_value(const std::pair<int, int>& pair) {
  static const std::size_t seed = tools::now();
  std::size_t res = seed;
  tools::hash_combine(res, pair.first);
  tools::hash_combine(res, pair.second);
  return res;
}
```
